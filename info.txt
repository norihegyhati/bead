    };

    function collision(head, array){
		for(let i = 0; i < array.length; i++)
		{
			if(head.x == array[i].x && head.y == array[i].y)
			{
				return true;
			}
		}
		return false;    
    }
    
    if (kigyoX < kp || kigyoY < kp ||
        kigyoX > ((meret - 1) * kp) || kigyoY > ((meret - 1) * kp) ||
        collision(newHead, kigyo)) {
        clearInterval(game);
    }

    kigyo.unshift(newHead);
